intervalMean<-aggregate(steps,by= list(interval=interval), FUN = mean, na.rm=T)
for(i in 1:length(steps)){
        if(is.na(steps[i])){
                NewActivity$steps[i]<-intervalMean$x[intervalMean$interval==interval[i]]
        }
}
